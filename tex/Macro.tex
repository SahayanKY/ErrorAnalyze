%微小量
\renewcommand{\d}{{\ensuremath{\rm d}}}


%偏微分
\newcommand{\pard}[3]{{%
	\ifnum #2<1%
		\text{エラー}%
	\else{%
		\ifnum #2=1%
			\ensuremath{%
				\frac{\partial #3}{\partial #1}%
				%\partial_{#1} #3%
			}%
		\else{%
			\ensuremath{%
				\frac{\partial^{#2} #3}{\partial #1^{#2}}%
				%\partial_{#1}^{#2} #3
			}%
		}\fi%
	}\fi%
}}
